VirusTotal API Integration

Python VirusTotal Lookup Function 
import requests
import hashlib
from datetime import datetime

class VirusTotalClient:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://www.virustotal.com/api/v3"
        self.headers = {"x-apikey": api_key}
    
    def check_hash(self, file_hash):
        """Check file hash against VirusTotal"""
        url = f"{self.base_url}/files/{file_hash}"
        response = requests.get(url, headers=self.headers)
        
        if response.status_code == 200:
            data = response.json()
            stats = data["data"]["attributes"]["last_analysis_stats"]
            return {
                "hash": file_hash,
                "malicious": stats["malicious"],
                "suspicious": stats["suspicious"],
                "harmless": stats["harmless"],
                "undetected": stats["undetected"],
                "detection_rate": f"{stats['malicious']}/{sum(stats.values())}",
                "verdict": "malicious" if stats["malicious"] > 3 else "clean"
            }
        elif response.status_code == 404:
            return {"hash": file_hash, "verdict": "unknown"}
        return None
    
    def check_ip(self, ip_address):
        """Check IP reputation"""
        url = f"{self.base_url}/ip_addresses/{ip_address}"
        response = requests.get(url, headers=self.headers)
        
        if response.status_code == 200:
            data = response.json()
            attrs = data["data"]["attributes"]
            return {
                "ip": ip_address,
                "country": attrs.get("country", "Unknown"),
                "as_owner": attrs.get("as_owner", "Unknown"),
                "reputation": attrs.get("reputation", 0),
                "malicious_votes": attrs["last_analysis_stats"]["malicious"],
                "harmless_votes": attrs["last_analysis_stats"]["harmless"]
            }
        return None

# Usage example
vt = VirusTotalClient("your-api-key")
result = vt.check_hash("44d88612fea8a8f36de82e1278abb02f")
print(f"Detection: {result['detection_rate']}")

AlienVault OTX Integration

Python OTX Pulse Fetcher 
from OTXv2 import OTXv2
from datetime import datetime, timedelta

class OTXFetcher:
    def __init__(self, api_key):
        self.otx = OTXv2(api_key)
    
    def get_recent_pulses(self, days=7):
        """Fetch recent threat pulses"""
        since = datetime.now() - timedelta(days=days)
        pulses = self.otx.getall_iter(modified_since=since)
        
        results = []
        for pulse in pulses:
            indicators = []
            for indicator in pulse.get("indicators", []):
                indicators.append({
                    "type": indicator["type"],
                    "value": indicator["indicator"],
                    "created": indicator.get("created")
                })
            
            results.append({
                "name": pulse["name"],
                "description": pulse.get("description", ""),
                "author": pulse["author_name"],
                "created": pulse["created"],
                "tags": pulse.get("tags", []),
                "malware_families": pulse.get("malware_families", []),
                "adversary": pulse.get("adversary", ""),
                "indicator_count": len(indicators),
                "indicators": indicators[:100]  # Limit for processing
            })
        return results
    
    def extract_iocs_by_type(self, pulses, ioc_type):
        """Extract specific IOC types from pulses"""
        iocs = set()
        for pulse in pulses:
            for indicator in pulse.get("indicators", []):
                if indicator["type"] == ioc_type:
                    iocs.add(indicator["value"])
        return list(iocs)

# Extract IOCs for SIEM ingestion
otx = OTXFetcher("your-otx-key")
pulses = otx.get_recent_pulses(days=7)

# Get specific IOC types
ipv4_iocs = otx.extract_iocs_by_type(pulses, "IPv4")
domain_iocs = otx.extract_iocs_by_type(pulses, "domain")
hash_iocs = otx.extract_iocs_by_type(pulses, "FileHash-SHA256")

CISA KEV Integration

Python CISA KEV Feed Parser 
import requests
from datetime import datetime

def fetch_cisa_kev():
    """Fetch CISA Known Exploited Vulnerabilities catalog"""
    url = "https://www.cisa.gov/sites/default/files/feeds/known_exploited_vulnerabilities.json"
    response = requests.get(url)
    data = response.json()
    
    vulnerabilities = []
    for vuln in data["vulnerabilities"]:
        vulnerabilities.append({
            "cve_id": vuln["cveID"],
            "vendor": vuln["vendorProject"],
            "product": vuln["product"],
            "name": vuln["vulnerabilityName"],
            "date_added": vuln["dateAdded"],
            "due_date": vuln["dueDate"],
            "known_ransomware": vuln.get("knownRansomwareCampaignUse", "Unknown"),
            "notes": vuln.get("notes", "")
        })
    
    return {
        "catalog_version": data["catalogVersion"],
        "date_released": data["dateReleased"],
        "total_count": data["count"],
        "vulnerabilities": vulnerabilities
    }

def get_overdue_kev(vulnerabilities):
    """Find KEVs past their remediation due date"""
    today = datetime.now().date()
    overdue = []
    
    for vuln in vulnerabilities:
        due_date = datetime.strptime(vuln["due_date"], "%Y-%m-%d").date()
        if due_date < today:
            vuln["days_overdue"] = (today - due_date).days
            overdue.append(vuln)
    
    return sorted(overdue, key=lambda x: x["days_overdue"], reverse=True)

# Fetch and process
kev_data = fetch_cisa_kev()
print(f"Total KEV entries: {kev_data['total_count']}")

# Get overdue vulnerabilities
overdue = get_overdue_kev(kev_data["vulnerabilities"])
print(f"Overdue KEVs: {len(overdue)}")

Internal Intelligence Sources

The most valuable and reliable intelligence often comes from within your own organization. These sources provide context-specific, validated intelligence.

Internal Intelligence Types

Incident Response Findings

  • Confirmed IOCs from actual incidents
  • Attacker TTPs observed in your environment
  • Malware samples from incidents
  • Lateral movement patterns

Honeypot/Deception Data

  • Attack patterns against decoys
  • Credential abuse attempts
  • Scanning and enumeration TTPs
  • Attacker tooling identification

Telemetry Analysis

  • Baseline deviation detection
  • Anomalous behavior patterns
  • Failed attack attempts
  • Environmental trends

Intelligence Source Evaluation

Admiralty Rating System

Use standardized ratings to evaluate and communicate intelligence reliability.

Source Reliability Rating Information Credibility Rating
Completely reliable A Confirmed by other sources 1
Usually reliable B Probably true 2
Fairly reliable C Possibly true 3
Not usually reliable D Doubtful 4
Unreliable E Improbable 5
Cannot be judged F Cannot be judged 6

Example: "A-1" indicates completely reliable source with confirmed information. "C-3" indicates fairly reliable source with possibly true information.

Interview Preparation

Q: What OSINT sources do you use for threat intelligence?

Strong Answer: I leverage multiple OSINT sources based on the intelligence need. For IOC validation and file reputation, VirusTotal is my primary source - I use it to check suspicious hashes and URLs before blocking. AlienVault OTX provides community-contributed threat pulses that often contain IOCs from recent campaigns. For attack surface awareness, I use Shodan to identify exposed assets. URLhaus and MalwareBazaar from abuse.ch are excellent for malware-specific IOCs. I also monitor CISA advisories religiously, especially the KEV catalog which helps prioritize patching. The key is understanding each source's strengths - VirusTotal for reputation, OTX for campaign context, Shodan for external visibility. I typically correlate indicators across multiple sources before acting to reduce false positives.

Q: How do you evaluate the quality of threat intelligence?

Strong Answer: I use several criteria: First, source reliability - is it from a vetted provider, government agency, or internal IR? Second, timeliness - older IOCs have diminishing value as attackers rotate infrastructure. Third, context - raw IOCs without attack context are less actionable than intelligence that explains the "why" and "how". Fourth, relevance - does this threat target my industry or technology stack? Fifth, actionability - can I actually use this to create a detection or take defensive action? I also track false positive rates from different feeds to weight their reliability. For formal documentation, I sometimes use the Admiralty rating system to communicate confidence levels. The best intelligence passes multiple quality checks and correlates with other sources.

Q: How do you integrate threat intelligence into your SIEM?

Strong Answer: In Sentinel, I use the Threat Intelligence blade to ingest STIX/TAXII feeds from commercial providers like Recorded Future or open sources like OTX. The ThreatIntelligenceIndicator table stores these IOCs, and I create analytics rules that match indicators against log data - comparing firewall logs against malicious IPs, proxy logs against malicious domains, and endpoint logs against file hashes. For real-time matching, I use the TI map analytics templates that automatically correlate across data sources. I also maintain watchlists for high-priority IOCs requiring immediate attention. The key considerations are setting appropriate confidence score thresholds (I typically require 70%+), handling IOC expiration to prevent stale data, and tuning to reduce noise from broad indicators. I track TI-generated alerts separately to measure feed effectiveness.

Q: What's the value of internal intelligence vs external feeds?

Strong Answer: Internal intelligence is often more valuable because it's contextual and validated. IOCs from your own incident response are confirmed relevant to your environment and represent actual attacker activity you've faced. Honeypot data shows what attackers specifically target in your infrastructure. Baseline deviations from your telemetry reveal anomalies unique to your environment that external feeds can't provide. External feeds provide broader coverage - they warn about threats you haven't encountered yet - but require filtering for relevance. The best approach combines both: external feeds for awareness and proactive defense, internal intelligence for precision and context. I prioritize internal findings for high-severity detections and use external feeds for broader coverage and validation.
