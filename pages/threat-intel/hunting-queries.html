Threat Hunting Queries

Ready-to-use KQL hunting queries organized by MITRE ATT&CK tactic for Microsoft Sentinel and Defender.

Initial Access Hunts

T1566 - Phishing 
// Hunt: Email attachments leading to process execution
EmailAttachmentInfo
| where TimeGenerated > ago(7d)
| where FileType in ("exe", "dll", "scr", "bat", "ps1", "vbs", "js")
| join kind=inner (
    DeviceProcessEvents
    | where TimeGenerated > ago(7d)
    | where InitiatingProcessFileName in~ ("outlook.exe", "winword.exe", "excel.exe")
) on $left.SHA256 == $right.SHA256
| project TimeGenerated, RecipientEmailAddress, FileName, FileType, 
          DeviceName, ProcessCommandLine
T1078 - Valid Accounts (Anomalous Logons) 
// Hunt: First-time logon from new country
let UserCountryHistory = SigninLogs
| where TimeGenerated > ago(90d) and TimeGenerated < ago(1d)
| summarize Countries = make_set(LocationDetails.countryOrRegion) by UserPrincipalName;

SigninLogs
| where TimeGenerated > ago(1d)
| where ResultType == 0  // Successful
| extend Country = tostring(LocationDetails.countryOrRegion)
| join kind=inner UserCountryHistory on UserPrincipalName
| where Country !in (Countries)
| project TimeGenerated, UserPrincipalName, Country, Countries, IPAddress, AppDisplayName

Execution Hunts

T1059.001 - PowerShell 
// Hunt: Suspicious PowerShell patterns
DeviceProcessEvents
| where TimeGenerated > ago(24h)
| where FileName in~ ("powershell.exe", "pwsh.exe")
| where ProcessCommandLine has_any (
    "-nop", "-noni", "-ep bypass", "-exec bypass",
    "IEX", "Invoke-Expression", "Invoke-WebRequest",
    "DownloadString", "DownloadFile", "FromBase64String",
    "Net.WebClient", "Start-BitsTransfer", "Invoke-Mimikatz"
)
| project TimeGenerated, DeviceName, AccountName, ProcessCommandLine
| extend ThreatIndicator = case(
    ProcessCommandLine has "Mimikatz", "Credential Theft",
    ProcessCommandLine has "DownloadString", "Download Cradle",
    ProcessCommandLine has "FromBase64String", "Encoded Payload",
    "Suspicious Execution"
)
T1204 - User Execution (Office Macros) 
// Hunt: Office applications spawning suspicious children
DeviceProcessEvents
| where TimeGenerated > ago(24h)
| where InitiatingProcessFileName in~ ("winword.exe", "excel.exe", "powerpnt.exe", "outlook.exe")
| where FileName in~ ("cmd.exe", "powershell.exe", "wscript.exe", "cscript.exe", 
                       "mshta.exe", "certutil.exe", "regsvr32.exe", "rundll32.exe")
| project TimeGenerated, DeviceName, InitiatingProcessFileName, FileName, 
          ProcessCommandLine, AccountName

Persistence Hunts

T1053.005 - Scheduled Tasks 
// Hunt: Suspicious scheduled task creation
SecurityEvent
| where TimeGenerated > ago(7d)
| where EventID == 4698  // Scheduled task created
| parse EventData with * '' TaskName '' *
| parse EventData with * '' TaskContent '' *
| where TaskName !startswith "\\Microsoft\\"  // Exclude Microsoft tasks
| where TaskContent has_any ("powershell", "cmd", "wscript", "http", "ftp", ".exe")
| project TimeGenerated, Computer, SubjectUserName, TaskName
| summarize Count = count() by TaskName
| where Count < 5  // Rare tasks
T1547.001 - Registry Run Keys 
// Hunt: New registry persistence
DeviceRegistryEvents
| where TimeGenerated > ago(24h)
| where ActionType == "RegistryValueSet"
| where RegistryKey has_any (
    "CurrentVersion\\Run",
    "CurrentVersion\\RunOnce",
    "CurrentVersion\\RunServices",
    "CurrentVersion\\Policies\\Explorer\\Run"
)
| where RegistryValueData has_any (".exe", ".dll", ".bat", ".ps1", ".vbs")
| where RegistryValueData !has "Microsoft"  // Exclude common Microsoft entries
| project TimeGenerated, DeviceName, RegistryKey, RegistryValueName, RegistryValueData

Credential Access Hunts

T1003 - OS Credential Dumping 
// Hunt: LSASS access from unusual processes
DeviceEvents
| where TimeGenerated > ago(24h)
| where ActionType == "OpenProcessApiCall"
| where TargetProcessName =~ "lsass.exe"
| where InitiatingProcessFileName !in~ (
    "csrss.exe", "services.exe", "svchost.exe", "wininit.exe",
    "MsMpEng.exe", "MsSense.exe", "taskmgr.exe", "procexp.exe"
)
| project TimeGenerated, DeviceName, InitiatingProcessFileName, 
          InitiatingProcessCommandLine, AccountName
| summarize Count = count() by InitiatingProcessFileName
| order by Count asc
T1558.003 - Kerberoasting 
// Hunt: Kerberoasting activity
SecurityEvent
| where TimeGenerated > ago(24h)
| where EventID == 4769
| where TicketEncryptionType == "0x17"  // RC4
| where ServiceName !endswith "$"  // Exclude machine accounts
| where ServiceName != "krbtgt"
| summarize 
    RequestCount = count(),
    UniqueServices = dcount(ServiceName),
    Services = make_set(ServiceName)
    by IpAddress, AccountName, bin(TimeGenerated, 1h)
| where RequestCount > 10 or UniqueServices > 5

Lateral Movement Hunts

T1021 - Remote Services 
// Hunt: Unusual lateral movement patterns
SecurityEvent
| where TimeGenerated > ago(24h)
| where EventID == 4624
| where LogonType in (3, 10)  // Network and RDP
| where AccountName !endswith "$"  // Exclude machine accounts
| where IpAddress != "-" and IpAddress != "127.0.0.1"
| summarize 
    TargetCount = dcount(Computer),
    Targets = make_set(Computer),
    LogonTypes = make_set(LogonType)
    by AccountName, IpAddress, bin(TimeGenerated, 1h)
| where TargetCount > 5  // User accessing many systems
T1021.002 - SMB/Windows Admin Shares 
// Hunt: Admin share access
SecurityEvent
| where TimeGenerated > ago(24h)
| where EventID == 5140  // Network share accessed
| where ShareName in ("\\\\*\\ADMIN$", "\\\\*\\C$", "\\\\*\\IPC$")
| where SubjectUserName !endswith "$"
| summarize AccessCount = count(), UniqueShares = dcount(ShareName) 
    by SubjectUserName, IpAddress, Computer
| where AccessCount > 10

Command and Control Hunts

T1071 - Application Layer Protocol (Beaconing) 
// Hunt: Potential beaconing activity
DeviceNetworkEvents
| where TimeGenerated > ago(24h)
| where RemotePort in (80, 443)
| summarize 
    ConnectionCount = count(),
    BytesSent = sum(SentBytes),
    AvgTimeBetween = avg(datetime_diff('second', TimeGenerated, prev(TimeGenerated)))
    by DeviceName, RemoteUrl, bin(TimeGenerated, 1h)
| where ConnectionCount > 50  // High frequency
| where AvgTimeBetween between (55 .. 65) or  // ~60 second intervals
        AvgTimeBetween between (295 .. 305) or  // ~5 minute intervals  
        AvgTimeBetween between (895 .. 905)     // ~15 minute intervals
| project DeviceName, RemoteUrl, ConnectionCount, AvgTimeBetween
T1071.004 - DNS Tunneling 
// Hunt: DNS tunneling indicators
DnsEvents
| where TimeGenerated > ago(24h)
| extend SubdomainLength = strlen(tostring(split(Name, ".")[0]))
| where SubdomainLength > 30  // Unusually long subdomain
| summarize 
    QueryCount = count(),
    UniqueSubdomains = dcount(Name),
    AvgSubdomainLength = avg(SubdomainLength)
    by ClientIP, tostring(split(Name, ".")[-2]) // Group by domain
| where UniqueSubdomains > 100 or AvgSubdomainLength > 40
